3. simbora fazer a query, preparar o statement protegendo de SQL Injection e retornar no fim a 
execução desse PDO statement

 	public function atualizar(){
        $query = 'update tb_tarefas SET tarefa = :tarefa where id = :id'; 
        $stmt = $this->conexao->prepare($query);
        $stmt->bindValue(':tarefa', $this->tarefa->__get('tarefa'));
        $stmt->bindValue(':id', $this->tarefa->__get('id'));
        return $stmt->execute();
    }

* lembrando que o retorno de execute() será 1 pra 1 registro atualizado e n para n registros atualizados
